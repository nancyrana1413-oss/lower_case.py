def count_case(s):
    lower=0
    upper=0
    for ch in s:
        if ch. islower():
            lower+=1
        elif ch .isupper():
            upper+=1
    print("uppercase letter:",upper)
    print("lowercase letter:", lower)
text=(input("enter the string:"))
count_case(text)
